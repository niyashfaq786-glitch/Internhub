Paste these in Firebase Console → Firestore Database → Rules, then Publish.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null
        && exists(/databases/$(database)/documents/admins/$(request.auth.uid))
        && get(/databases/$(database)/documents/admins/$(request.auth.uid)).data.status == "ACTIVE";
    }

    // Admins: only admins can read/write
    match /admins/{uid} {
      allow read, write: if isAdmin();
    }

    // Students:
    // - Admin can read/write all students
    // - A logged-in student can read/write ONLY their own student document
    //   (students/{student_id} where resource.data.uid == request.auth.uid)
    match /students/{studentId} {
      allow read, write: if isAdmin()
        || (request.auth != null
            && resource.data.uid == request.auth.uid);
    }

    // Opportunities: public read, admin write
    match /opportunities/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Admin requests: anyone can create request; admin can read/update/delete
    match /admin_requests/{id} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }

    // Applications: (optional) keep admin-only for now
    match /applications/{id} {
      allow read, write: if isAdmin();
    }
  }
}
